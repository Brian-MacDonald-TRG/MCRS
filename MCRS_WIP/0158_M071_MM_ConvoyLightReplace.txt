			<proc>
				<step1>
					<specpara>
						<note>
							<trim.para>The convoy light pole is inserted and bolted to the TDRT at the driver side rear drawer. The pole can be removed and or mounted with the drawers secured in the normal or wide position.</trim.para>
						</note>
						<para>Remove two bolts on rubber clamp that secures the convoy light to the TDRT <callout assocfig="M00071-9-2320-452-F0001" label ="1"/>.</para>
					</specpara>
				</step1>

					<figure id="M00071-9-2320-452-F0001">
					<title>Convoy Light Pole Clamp</title>
					<graphic boardno="G06232000397"/>
					</figure>

				<step1>
					<para>Remove light pole from TDRT.</para>
				</step1>
				<step1>
					<para>Disconnect QD electrical connector at base of light pole <callout assocfig="M00071-9-2320-452-F0002" label ="1"/>.</para>
				</step1>

					<figure id="M00071-9-2320-452-F0002">
					<title>Disconnecting Convoy Light</title>
					<graphic boardno="G06232000398"/>
					</figure>

				<step1>
					<para>Bring the pole down to the TDRT deck and remove unserviceable convoy light from the pole by loosening wing nut and pulling unserviceable unit off the pole <callout assocfig="M00071-9-2320-452-F0003" label ="1"/>.</para>
				</step1>

					<figure id="M00071-9-2320-452-F0003">
					<title>Convoy Light Removal</title>
					<graphic boardno="G06232000399"/>
					</figure>

				<step1>
					<para>Obtain serviceable item.</para>
				</step1>
				<step1>
					<para>Push light onto pole completely and secure with wing nut.</para>
				</step1>
				<step1>
					<para>Reconnect QD electrical connector at base of light pole.</para>
				</step1>
				<step1>
					<para>Place rubber clamp on convoy light pole.</para>
				</step1>
				<step1>
					<para>Insert pole into opening in TDRT drawer</para>
				</step1>
				<step1>
					<para>Install 2 bolts through rubber clamp to secure convoy light to the TDRT</para>
				</step1>
			</proc>
